Instance internalize_P : computable P.
Proof.
  change P with
      (fun (R : SRS) (x0 y0 : Definitions.string) =>
         [d R x0 y0; e R x0 y0] ++ R ++ [[hash R x0 y0] / [hash R x0 y0]] ++ map singcard (x0 ++ y0 ++ sym R)).
  extract.
Defined.
