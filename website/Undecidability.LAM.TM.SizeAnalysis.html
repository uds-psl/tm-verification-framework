Fixpoint sum (A:list nat) :=
  match A with
    [] => 0
  | a::A => a + sum A
  end.

Lemma sum_app A B : sum (A++B) = sum A + sum B.
Proof.
  induction A;cbn;omega.
Qed.

Hint Rewrite sum_app : list.

Definition sizeT t :=
    match t with
      varT n => 1 + n
    | _ => 1
    end.

Definition sizeP (P:Pro) := 1 + sum (map sizeT P).
Hint Unfold sizeP.
